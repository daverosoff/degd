\noindent This quiz is \emph{open-note}, but no books or calculators.

\begin{questions}

\question Consider the function $f(y) = y(1-y^2)$. Sketch the graph of $f(y)$ versus $y$ and classify the equilibrium points of the differential equation [ y’ = y(1-y^2), \quad -\infty < y_0 < \infty ] as stable, unstable, or semistable. Sketch the graphs of several solutions in the $(t,y)$-plane.

\dwrspace{1}

\question Solve the exact differential equation [ (3x^2 - 2xy + 2) + (6y^2 - x^2 + 3) \frac{dy}{dx} = 0. ] Your solution should involve an arbitrary constant.

\dwrspace{1}

\end{questions}
