\subsection*{Reduction of order}

Reduction of order is a general method for finding a second solution
to a second-order linear differential equation if one solution is
already known. 

\medlines

\begin{questions}

    \question Verify that $y_1(t) = t^2$ is a solution of the
    differential equation $t^2 y'' - 4ty' + 6y = 0$, $t > 0$.

    \dwrspace{1} 

    \question Use reduction of order to find a second solution.

    \dwrspace{6}

    \newpage

    \question Use undetermined coefficients to find particular
    solutions of:
        \begin{parts}
            \part $y'' - 3y' - 4y = 3e^{2t}$

            \dwrspace{1}

            \part $y'' - 3y' - 4y = 2 \sin{t}$

            \dwrspace{1}

            \part $y'' + 10y' + 16y = 14e^{-t}$

            \dwrspace{1}

        \end{parts}
    
\end{questions}
